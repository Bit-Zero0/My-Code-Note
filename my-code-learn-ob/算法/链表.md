
# 合并两个有序列表
https://leetcode.cn/problems/merge-two-sorted-lists/description/

## 解题思路
创建一个带头结点的链表，比较list1 和list2每个节点的值，较小的插入到我们创建的链表中。两条链表最终有可能还有节点还没被插入到我们创建的列表中，所以还需要最后一层判断。


## C++题解
```c
//带哨兵位的头结点  
class Solution {  
    public:  
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {  
        ListNode* newList = new ListNode(-1);  //创建头结点
        ListNode* tmp = newList;  
  
        while(list1 && list2)  
        {  
            if(list1->val < list2->val)  
            {  
                tmp->next = list1;  
                list1 = list1->next;  
                tmp = tmp->next;  
            }  
            else  
            {  
                tmp->next = list2;  
                list2 = list2->next;  
                tmp = tmp->next;  
            }  
        }  
  
        if(list1)  //list1 还有节点未被插入
            tmp->next = list1;  
  
        if(list2)  //list2 还有节点未被插入
            tmp->next = list2;  
  
        ListNode* next = newList->next;  
        delete newList;  
  
        return next;  
    }  
};
```

```java
            tmp.next = list2;  
  
        return NewList.next;  
    }  
}
```

# 反转链表
 [206. 反转链表](https://leetcode.cn/problems/reverse-linked-list/)

## 解题思路
使用双指针，cur比prev先一步，在将cur的next指向prev即可。注意保存cur的next。

## C++题解
```java
class Solution {  
    public:  
    ListNode* reverseList(ListNode* head) {  
        if(head == nullptr)return nullptr;  
  
        ListNode* prev = nullptr;  
        ListNode* cur = head;  
  
        while(cur != nullptr){  
            ListNode* next = cur->next;  //先保存cur的下一个节点
            cur->next = prev;  
            prev = cur;  
            cur = next;  
        }  
        return prev;  
    }  
};
```

## Java题解
```java
class Solution {

            cur = next;  
        }  
        return prev;  
    }
```

# 移除链表元素
[203. 移除链表元素](https://leetcode.cn/problems/remove-linked-list-elements/)

## C++题解
```cpp
class Solution {  
    public:  
    ListNode* removeElements(ListNode* head, int val) {  
        if(head == nullptr)return nullptr;  
  
        ListNode* prev = head;  
        ListNode* cur = head->next;  
        while(cur != nullptr)  
        {  
            if(cur->val == val)  
            {  
                prev->next = cur->next;  
                cur = cur->next;  
            }  
            else  
            {  
                prev = cur ;  
                cur = cur->next;  
            }  
        }  
        if(head->val == val)  
        {  
            head = head->next;  
        }  
        return head;  
    }  
};
```

# 链表的中间结点
[876. 链表的中间结点](https://leetcode.cn/problems/middle-of-the-linked-list/)

## 解题思路
使用快慢指针，fast指针的速度是slow指针的两倍。

## C++题解
```cpp
class Solution {  
    public:  
    ListNode* middleNode(ListNode* head) {  
        if(head == nullptr) return nullptr;  
        ListNode* fast = head;  
        ListNode* slow = head;  
        while(fast != nullptr && fast->next != nullptr){  
            fast = fast->next->next;  
            slow = slow->next;  
        }  
        return slow;  
    }  
};
```


## java题解

# 链表分割
https://www.nowcoder.com/practice/0e27e0b064de4eacac178676ef9c9d70?

## 解题思路
创建四个指针 ，`as` ， `ae` ， `bs`  ，`be` ,  (as = a start , ae = a end...以此类推),使用as到ae来操作小于x的数，使用bs到be来操作大于x的数。

## C++题解
```cpp
class Partition {  
    public:  
    ListNode* partition(ListNode* pHead, int x) {  
        // write code here  
        if(pHead == nullptr) return nullptr;  
  
        ListNode* as = nullptr;  
        ListNode* ae = nullptr;  
        ListNode* bs = nullptr;  
        ListNode* be = nullptr;  
        ListNode* cur = pHead;  
        while(cur != nullptr)  
        {  
            if(cur->val < x )  
            {  
                if(ae == nullptr)  
                {  
                    as = cur;  
                    ae = cur;  
                }  
                else  
                {  
                    ae->next = cur;  
                    ae = ae->next;  
                }  
            }  
            else  
            {  
                if(be == nullptr)  
                {  
                    bs = cur;  
                    be = cur;  
                }  
                else  
                {  
                    be->next = cur;  
                    be = be->next;  
                }  
            }  
            cur = cur->next;  
        }  
        if(as == nullptr)  
        {  
            return bs;  
        }  
        ae->next = bs;  
  
        if(bs != nullptr)  
        {  
            be->next = nullptr;  
        }  
  
        return as;  
  
    }  
};
```

## Java题解
```java
public class Partition {  
    public ListNode partition(ListNode pHead, int x) {  
        ListNode bs = null;  
        ListNode be = null;  
        ListNode as = null;  
        ListNode ae = null;  
        ListNode cur = pHead;  
        while (cur != null) {  
            if (cur.val < x) {  
                //1、第一次  
                if (as == null) {  
                    as = cur;  
                    ae = cur;  
                } else {  
                    //2、不是第一次  尾插  
                    ae.next = cur;  
                    ae = ae.next;  
                }  
            } else {  
                if (bs == null) {  
                    bs = cur;  
                    be = cur;  
                } else {  
                    //2、不是第一次  
                    be.next = cur;  
                    be = be.next;  
                }  
            }  
            cur = cur.next;  
        }  
        //预防第一个段为空  
        if (as == null) {  
            return bs;  
        }  
        ae.next = bs;  
        //预防第2个段当中的数据，最后一个节点不是空的  
        if (bs != null) {  
            be.next = null;  
        }  
        return as;  
    }  
}

```



# 链表中倒数第k个结点
https://www.nowcoder.com/practice/529d3ae5a407492994ad2a246518148a

## 解题思路
本题使用的是快慢指针，既然我们要找倒数第k个节点，我们就让快指针先走k-1步(要注意k-1是否超过链表长度)，随后快慢指针一起行走，当快指针到达终点时，慢指针就是我们要找的倒数第K个节点。

## C++题解
```cpp
class Solution {  
    public:  
    ListNode* FindKthToTail(ListNode* pListHead, unsigned int k) {  
        if(k < 0 || pListHead  == nullptr) return nullptr;  
        ListNode* prev = pListHead;  
        ListNode* cur = pListHead;  
  
        while(k -1 != 0)  
        {  
            prev = prev->next;  
            if(prev == nullptr) return nullptr;  //判断k是否超过链表的长度
            k--;  
        }  
  
        while(prev->next != nullptr)  
        {  
            prev = prev->next;  
            cur = cur->next;  
        }  
  
        return cur;  
    }  
};
```


## java题解
```java

    public ListNode FindKthToTail(ListNode head,int k) {

        ListNode prev = head;

        }  
        return cur;  
    }
```

# 删除链表中重复的结点
https://www.nowcoder.com/practice/fc533c45b73a41b0b44ccba763f866ef?


## C++题解
```cpp
class Solution {  
    public:  
    ListNode* deleteDuplication(ListNode* pHead) {  
        if(pHead == nullptr) return nullptr;  
        ListNode* newList = new ListNode(-1);  
        ListNode* tmp = newList;  
        ListNode* cur = pHead;  
        while(cur != nullptr){  
            if(cur->next != nullptr && cur->val == cur->next->val ){  
                while(cur->next != nullptr && cur->val == cur->next->val){  
                    cur = cur->next;  
                }  
  
                cur = cur->next;//多走一步  
            }else{  
                tmp->next = cur;  
                tmp = tmp->next;  
                cur = cur->next;  
            }  
        }  
  
        tmp->next = nullptr;//防止最后一个节点的值也是重复的  
        ListNode* next  = newList->next;  
        delete newList;  
        return next;  
    }  
};
```

# 回文链表
 [234. 回文链表](https://leetcode.cn/problems/palindrome-linked-list/)
 [牛客网 回文链表](https://www.nowcoder.com/practice/d281619e4b3e4a60a2cc66ea32855bfa?)

## 解题思路
先找到中间节点，如果链表的节点是偶数个 ，则找到中间的第二个节点，从此节点开始反转链表，最后使用双指针进行判断即可。


## C++ 题解
不知为何在leetcode中无法通过，但是牛客网没问题。
```cpp
class PalindromeList {  
    public:  
    bool chkPalindrome(ListNode* head) {  
        // write code here  
        if(head == nullptr) return true;  
        ListNode* fast = head;  
        ListNode* slow = head;  
  
        while(fast != nullptr && fast->next != nullptr)  
        {  
            fast = fast->next->next;  
            slow = slow->next;  
        }  
        //slow走到了中间位置->反转 
  
        ListNode* cur = slow->next;  
        while(cur != nullptr)  
        {  
            ListNode* next = cur->next;  
            cur->next = slow;  
            slow = cur;  
            cur = next;  
        }  
		//反转完成
        while(head != slow)  
        {  
            if(head->val != slow->val)  
                return false;  
  
            if(head->next == slow)  
                return true;  
  
            head = head->next;  
            slow = slow->next;  
        }  
        return true;  
    }  
};
```

```cpp
class Solution {  
public:  
    bool isPalindrome(ListNode* head) {  
        std::vector<int> vec;  
        while(head != nullptr)  
        {  
            vec.push_back(head->val);  
            head = head->next;  
        }  
  
        for( int i = 0 , j = vec.size() -1 ;i < j ; i++ ,j--)  
        {  
            if(vec[i] != vec[j]) return false;  
        }  
        return true;  
    }  
};
```
