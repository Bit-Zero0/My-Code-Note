class Solution {
public:
    int minOperations(vector<int>& nums, int x) {
        int sum = 0;
        for(auto i : nums)
            sum += i;
       
        int target = sum - x;
        if( target < 0) return -1; // 当整个数组的值之和小于x时返回-1

        sum = 0;
        int len = -1;
        for( int left = 0 , right = 0; right< nums.size(); right++)
        {
            sum += nums[right];
            while(sum > target)
                sum -= nums[left++];
            if(sum == target)
                len = max(len , right- left +1);
        }
        if(len == -1)
            return -1;
        
        return  nums.size() - len;
    }
}; ^rdkJjX5h
